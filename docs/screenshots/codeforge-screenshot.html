async function fetchUserData(userId: string): Promise<User> { const maxRetries = 3; for (let i = 0; i < maxRetries; i++) { try { const res = await fetch(`/api/users/${userId}`); if (!res.ok) throw new Error(`HTTP ${res.status}`); return await res.json(); } catch (err) { if (i === maxRetries - 1) throw err; await delay(1000 * (i + 1)); } } }
